\def\@makechapterhead#1
  {
    \vspace*{-5\p@}%
      {
        \parindent \z@ \centering \sffamily
          \large \thechapter \quad  #1
          \par\nobreak
          \vskip 26\p@ \par\nobreak
      }
    \thispagestyle{fancy}
  }

\renewcommand\section
  {\vspace*{-2.3ex}
    \@startsection {section}{1}{\z@}%
      {0ex}%
      {.1ex}%
      {\sffamily\large}
  }
