\def\@makechapterhead#1
  {
    \vspace*{-13\p@}%
      {
        \parindent \z@ \centering \sffamily
          \large \thechapter \quad  #1
          \par\nobreak
          \vskip 25\p@ \par\nobreak
      }
    \thispagestyle{fancy}
  }
\def\@makeschapterhead#1
  {
    \vspace*{-13\p@}%
      {
        \parindent \z@ \raggedright
          \centering \sffamily
          \interlinepenalty\@M
          \large \bfseries #1\par\nobreak
          \vskip 25\p@
      }
    \thispagestyle{fancy}
  }
